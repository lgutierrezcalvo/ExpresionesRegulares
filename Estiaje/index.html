//Devuelve (Sin datos) si algun campo no tiene datos.
if ($feature.Octubre <= 0 || $feature.Noviembre <= 0 || $feature.Diciembre <= 0 || $feature.Enero <= 0 || $feature.Febrero <= 0 || $feature.Marzo <= 0 || $feature.Abril <= 0 || $feature.Mayo <= 0 || $feature.Junio <= 0 || $feature.Julio <= 0 || $feature.Agosto <= 0 || $feature.Septiembre <= 0) {
return '(Sin datos)';
};

//Bucle para obtener el valor mas bajo de cada entidad y relacionarlo con su mes correspondiente.
var array = [$feature.Octubre, $feature.Noviembre, $feature.Diciembre, $feature.Enero, $feature.Febrero, $feature.Marzo, $feature.Abril, $feature.Mayo, $feature.Junio, $feature.Julio, $feature.Agosto, $feature.Septiembre]; var nombreMeses = ['Octubre 2014','Noviembre 2014','Diciembre 2014','Enero 2015','Febrero 2015','Marzo 2015','Abril 2015','Mayo 2015','Junio 2015','Julio 2015','Agosto 2015','Septiembre 2015'];
var minimo = 9999;
var mes = "";
for (var arrayIndex = 0; arrayIndex < 12; arrayIndex++) {
if (array[arrayIndex] < minimo) {
    minimo = array[arrayIndex];
    mes = nombreMeses[arrayIndex];
};
};

//Devuelve el mes de mayor caudal en cada estación de aforo, seguido del valor correspondiente a ese mes, acompañado de la unidad.
return mes + ' - ' + minimo + ' m3/s';
